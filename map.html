let map, trafficMap; let syncing = false; function initMap() { const centerCoords = { lat: 14.8370, lng: 120.8856 }; // Guiguinto, Bulacan // Plain map (background) map = new google.maps.Map(document.getElementById("map"), { center: centerCoords, zoom: 12, mapTypeId: 'roadmap', // Default to 'roadmap' view disableDefaultUI: true, // Disable all default UI controls }); new google.maps.Marker({ position: centerCoords, map: map, title: "Plain Map", }); // Circular traffic map (center overlay) trafficMap = new google.maps.Map(document.getElementById("traffic-map"), { center: centerCoords, zoom: 12, mapTypeId: 'roadmap', // Default to 'roadmap' view disableDefaultUI: true, // Disable all default UI controls disableDefaultUI: true, }); const trafficLayer = new google.maps.TrafficLayer(); trafficLayer.setMap(trafficMap); // Synchronize both maps syncMaps(map, trafficMap); syncMaps(trafficMap, map); } function syncMaps(source, target) { source.addListener("center_changed", () => { if (syncing) return; syncing = true; target.setCenter(source.getCenter()); target.setZoom(source.getZoom()); syncing = false; }); }
